Catalog#

Interface#

class Catalog#

This interface is responsible for reading and writing metadata such as database/table from a paimon catalog.

Public Functions

virtual Status CreateDatabase(const std::string &name, const std::map<std::string, std::string> &options, bool ignore_if_exists) = 0#

Creates a database with the specified properties.

Parameters:

  • name – Name of the database to be created.

  • options – Additional properties associated with the database.

  • ignore_if_exists – If true, no action is taken if the database already exists. If false, an error status is returned if the database exists.

Returns:

A status indicating success or failure.

virtual Status CreateTable(const Identifier &identifier, ArrowSchema *c_schema, const std::vector<std::string> &partition_keys, const std::vector<std::string> &primary_keys, const std::map<std::string, std::string> &options, bool ignore_if_exists) = 0#

Creates a new table in the catalog.

Note

System tables cannot be created using this method.

Parameters:

  • identifier – Identifier of the table to be created.

  • c_schema – The schema of the table to be created.

  • partition_keys – List of columns that should be used as partition keys for the table.

  • primary_keys – List of columns that should be used as primary keys for the table.

  • options – Additional table-specific options.

  • ignore_if_exists – If true, no action is taken if the table already exists. If false, an error status is returned if the table exists.

Returns:

A status indicating success or failure.

virtual Result<std::vector<std::string>> ListDatabases() const = 0#

Lists all the databases available in the catalog.

Returns:

A result containing a vector of database names, or an error status.

virtual Result<std::vector<std::string>> ListTables(const std::string &db_name) const = 0#

Lists all the tables within a specified database.

Note

System tables will not be listed.

Parameters:

db_name – The name of the database to list tables from.

Returns:

A result containing a vector of table names in the specified database, or an error status.

virtual Status DropDatabase(const std::string &name, bool ignore_if_not_exists, bool cascade) = 0#

Drops a database.

Parameters:

  • name – Name of the database to be dropped.

  • ignore_if_not_exists – If true, no action is taken if the database does not exist.

  • cascade – If true, drops all tables and functions in the database before dropping the database.

Returns:

A status indicating success or failure.

virtual Status DropTable(const Identifier &identifier, bool ignore_if_not_exists) = 0#

Drops a table.

Parameters:

  • identifier – Identifier of the table to drop.

  • ignore_if_not_exists – If true, no action is taken if the table does not exist.

Returns:

A status indicating success or failure.

virtual Status RenameTable(const Identifier &from_table, const Identifier &to_table, bool ignore_if_not_exists) = 0#

Renames a table.

Parameters:

  • from_table – Current identifier of the table.

  • to_table – New identifier for the table.

  • ignore_if_not_exists – If true, no action is taken if the table does not exist.

Returns:

A status indicating success or failure.

virtual Result<std::shared_ptr<Table>> GetTable(const Identifier &identifier) const = 0#

Gets a table.

Parameters:

identifier – Identifier of the table to get.

Returns:

A result containing the table, or an error status.

virtual Result<bool> DatabaseExists(const std::string &db_name) const = 0#

Checks whether a database with the specified name exists in the catalog.

Parameters:

db_name – The name of the database to check for existence.

Returns:

A result containing true if the database exists, false otherwise, or an error status.

virtual Result<bool> TableExists(const Identifier &identifier) const = 0#

Checks whether a table with the specified identifier exists in the catalog.

Parameters:

identifier – The identifier of the table to check for existence.

Returns:

A result containing true if the table exists, false otherwise, or an error status.

virtual std::string GetDatabaseLocation(const std::string &db_name) const = 0#

Returns the expected location of a specified database.

Note

This does not check whether the database actually exists.

Parameters:

db_name – The name of the database to get the location for.

Returns:

A string representing the expected location of the database.

virtual std::string GetTableLocation(const Identifier &identifier) const = 0#

Returns the expected location of a specified table.

Note

This does not check whether the table actually exists.

Parameters:

identifier – The table identifier containing database and table name.

Returns:

A string representing the expected location of the table.

virtual std::string GetRootPath() const = 0#

Returns the root path of the catalog.

Returns:

A string representing the root path of the catalog.

virtual std::shared_ptr<FileSystem> GetFileSystem() const = 0#

Returns the file system used by the catalog.

Returns:

A shared pointer to the file system instance.

virtual Result<std::shared_ptr<Schema>> LoadTableSchema(const Identifier &identifier) const = 0#

Loads the latest schema of a specified table.

Note

System tables will not be supported.

Parameters:

identifier – The identifier (database and table name) of the table to load.

Returns:

A result containing table schema if the table exists, or an error status on failure.

virtual Result<std::vector<SnapshotInfo>> ListSnapshots(const Identifier &identifier, const std::string &branch = "") const = 0#

Lists all snapshots of the specified table, ordered by snapshot id.

Parameters:

  • identifier – The identifier (database and table name) of the table.

  • branch – Branch name; empty string means the main branch.

Returns:

A result containing a vector of SnapshotInfo ordered by snapshot id ascending, or an error status.

Public Static Functions

static Result<std::unique_ptr<Catalog>> Create(const std::string &root_path, const std::map<std::string, std::string> &options, const std::shared_ptr<FileSystem> &file_system = nullptr)#

Factory method for creating a Catalog instance.

Parameters:

  • root_path – Path to the root directory where the catalog is located.

  • options – Configuration options for catalog initialization.

  • file_system – Specifies the file system for file operations. If not set, use default file system (configured in Options::FILE_SYSTEM)

Returns:

A result containing a unique pointer to a Catalog instance, or an error status.
